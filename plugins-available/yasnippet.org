#+begin_src emacs-lisp
  (dolist (package '(yasnippet))
    (when (not (package-installed-p package))
      (package-install package)))
  (require 'yasnippet)
  (yas-global-mode 1)
  (setq yas-snippet-dirs '("~/.emacs.d/snippets"))
#+end_src
